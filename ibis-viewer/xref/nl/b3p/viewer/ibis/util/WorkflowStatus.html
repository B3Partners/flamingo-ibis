17  package nl.b3p.viewer.ibis.util;
18  
19  /**
20   * mimic WorkflowStatus.js, specifically the the status.
21   *
22   * @author Mark Prins <mark@b3partners.nl>
23   */
24  public enum WorkflowStatus {
25  
26      // explicitly using lowercase, to mimic the javascript
27      nieuw("Nieuw"),
28      beoordeling_gemeente("Beoordeling gemeente"),
29      goedkeuring_gemeente("Goedkeuring gemeente"),
30      goedkeuring_provincie("Goedkeuring provincie"),
31      definitief("Definitief"),
32      archief("Archief");
33  
34      private final String label;
35  
36      /**
37       * The name of the workflow attribute field in the datamodel {@value }.
38       */
39      public static final String workflowFieldName = "workflow_status";
40  
41      WorkflowStatus(String label) {
42          this.label = label;
43      }
44  
45      public String label() {
46          return this.label;
47      }
48  
49  }
