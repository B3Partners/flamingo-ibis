70      @Override
71      protected String addNewFeature() throws Exception {
72          String kavelID = super.addNewFeature();
73          //update  terrein
74          Object terreinID = this.getJsonFeature().optString(KAVEL_TERREIN_ID_FIELDNAME, null);
75          WorkflowStatus status = WorkflowStatus.valueOf(this.getJsonFeature().optString(WORKFLOW_FIELDNAME, WorkflowStatus.bewerkt.name()));
76          if (terreinID != null) {
77              WorkflowUtil.updateTerreinGeometry(Integer.parseInt(terreinID.toString()), this.getLayer(), status, this.getApplication(), Stripersist.getEntityManager());
78          }
79          return kavelID;
80      }

82      /**
83       * Override to not delete a feature but set workflow status to
84       * {@code WorkflowStatus.archief}
85       *
86       * @param fid feature id
87       * @throws IOException if any
88       * @throws Exception if any
89       * @see WorkflowStatus
90       */
91      @Override
92      protected void deleteFeature(String fid) throws IOException, Exception {
93          if(this.isHistorisch()){
94              this.deleteFeatureHistorisch(fid);
95          } else {
96              log.debug("ibis deleteFeature: " + fid);
97  
98              Transaction transaction = new DefaultTransaction("ibis_delete");
99              this.getStore().setTransaction(transaction);
100 
101             FilterFactory2 ff = CommonFactoryFinder.getFilterFactory2();
102             Filter filter = ff.id(new FeatureIdImpl(fid));
103 
104             try {
105                 this.getStore().modifyFeatures(WORKFLOW_FIELDNAME, WorkflowStatus.afgevoerd, filter);
106                 SimpleFeature original = this.getStore().getFeatures(filter).features().next();
107                 transaction.commit();
108                 Object terreinID = original.getAttribute(KAVEL_TERREIN_ID_FIELDNAME);
109                 if (terreinID != null) {
110                     WorkflowUtil.updateTerreinGeometry(Integer.parseInt(terreinID.toString()), this.getLayer(), WorkflowStatus.afgevoerd, this.getApplication(), Stripersist.getEntityManager());
111                 }
112             } catch (Exception e) {
113                 transaction.rollback();
114                 throw e;
115             } finally {
116                 transaction.close();
117             }
118         }
119     }

121     /**
122      * verwijder historische record.
123      * Als het een "archief" record is dan gewoon verwijderen,
124      * als het een "afgevoerd" record is dan verwijderen en jongste "archief" record "afgevoerd" maken.
125      * als er geen "archief" te vinden is dan laten.
126      *
127      * @param fid feature to remove
128      * @throws IOException if any
129      * @throws Exception if any
130      */
131     private void deleteFeatureHistorisch(String fid) throws IOException, Exception {
132         log.debug("ibis deleteFeatureHistorisch: " + fid);
133 
134         Transaction transaction = new DefaultTransaction("ibis_delete_historisch");
135         this.getStore().setTransaction(transaction);
136 
137         FilterFactory2 ff = CommonFactoryFinder.getFilterFactory2();
138         Filter fidFilter = ff.id(new FeatureIdImpl(fid));
139 
140         Filter hist = ff.or(
141                 ff.equal(ff.property(WORKFLOW_FIELDNAME), ff.literal(WorkflowStatus.archief.name()), false),
142                 ff.equal(ff.property(WORKFLOW_FIELDNAME), ff.literal(WorkflowStatus.afgevoerd.name()), false)
143         );
144 
145         try {
146             SimpleFeature original = this.getStore().getFeatures(fidFilter).features().next();
147             Number ibisId = (Number) original.getAttribute(ID_FIELDNAME);
148             Object terreinID = original.getAttribute(KAVEL_TERREIN_ID_FIELDNAME);
149 
150             log.debug(String.format("Delete from feature source #%d fid=%s",
151                     this.getLayer().getFeatureType().getId(), fid));
152 
153             switch (WorkflowStatus.valueOf( original.getAttribute(WORKFLOW_FIELDNAME).toString()) ) {
154                 case archief:
155                     // archief record gewoon verwijderen
156                     this.getStore().removeFeatures(fidFilter);
157                     break;
158                 case afgevoerd:
159                     // zoek alle historische kavels voor dit kavel nummer
160                     SimpleFeatureCollection histRecords = this.getStore().getFeatures(
161                             ff.and(
162                                     hist,
163                                     ff.equals(ff.property(ID_FIELDNAME), ff.literal(ibisId))
164                             ));
165                     if (histRecords.size() > 1) {
166                         // als er naast deze afgevoerd nog andere zijn
167                         histRecords.sort(ff.sort(MUTATIEDATUM_FIELDNAME, SortOrder.DESCENDING));
168                         // dan de jongste daarvan (eerste uit de lijst) op afgevoerd zetten
169                         String updateFid = histRecords.features().next().getID();
170                         this.getStore().modifyFeatures(WORKFLOW_FIELDNAME, WorkflowStatus.afgevoerd, ff.id(new FeatureIdImpl(updateFid)));
171                         // en gevraagde verwijderen uit database
172                         this.getStore().removeFeatures(fidFilter);
173                     }
174                     break;
175                 default:
176                     throw new UnsupportedOperationException("Historische record met status anders dan 'archief' of 'afgevoerd' kan niet worden verwijderd.");
177             }
178             transaction.commit();
179             // update terrein van kavel
180             if (terreinID != null) {
181                 WorkflowUtil.updateTerreinGeometry(Integer.parseInt(terreinID.toString()), this.getLayer(), WorkflowStatus.afgevoerd, this.getApplication(), Stripersist.getEntityManager());
182             }
183         } catch (Exception e) {
184             transaction.rollback();
185             throw e;
186         } finally {
187             transaction.close();
188         }
189

192     /**
193      * Override the method from the base class to process our workflow.
194      *
195      * @param fid feature id
196      * @throws Exception if any
197      */
198     @Override
199     protected void editFeature(String fid) throws Exception {
200         log.debug("ibis editFeature:" + fid);
201         FilterFactory2 ff = CommonFactoryFinder.getFilterFactory2();
202         Filter fidFilter = ff.id(new FeatureIdImpl(fid));
203 
204         List<String> attributes = new ArrayList();
205         List values = new ArrayList();
206         WorkflowStatus incomingWorkflowStatus = null;
207         // parse json
208         for (Iterator<String> it = this.getJsonFeature().keys(); it.hasNext();) {
209             String attribute = it.next();
210             if (!this.getFID().equals(attribute)) {
211                 AttributeDescriptor ad = this.getStore().getSchema().getDescriptor(attribute);
212                 if (ad != null) {
213                     if (!isAttributeUserEditingDisabled(attribute)) {
214                         attributes.add(attribute);
215                         if (ad.getType() instanceof GeometryType) {
216                             String wkt = this.getJsonFeature().getString(ad.getLocalName());
217                             Geometry g = null;
218                             if (wkt != null) {
219                                 g = new WKTReader().read(wkt);
220                             }
221                             values.add(g);
222                         } else {
223                             String v = this.getJsonFeature().getString(attribute);
224                             values.add(StringUtils.defaultIfBlank(v, null));
225                             // remember the incoming workflow status
226                             if (attribute.equals(WORKFLOW_FIELDNAME)) {
227                                 incomingWorkflowStatus = WorkflowStatus.valueOf(v);
228                             }
229                         }
230                     } else {
231                         log.info(String.format("Attribute \"%s\" not user editable; ignoring", attribute));
232                     }
233                 } else {
234                     log.warn(String.format("Attribute \"%s\" not in feature type; ignoring", attribute));
235                 }
236             }
237         }
238         log.debug(String.format("Modifying feature source #%d fid=%s, attributes=%s, values=%s",
239                 this.getLayer().getFeatureType().getId(), fid, attributes.toString(), values.toString()));
240 
241         Transaction editTransaction = new DefaultTransaction("ibis_edit");
242         this.getStore().setTransaction(editTransaction);
243         try {
244             if (incomingWorkflowStatus == null) {
245                 throw new IllegalArgumentException("Workflow status van edit feature is null, dit wordt niet ondersteund.");
246             }
247             SimpleFeature original = this.getStore().getFeatures(fidFilter).features().next();
248             Object terreinID = original.getAttribute(KAVEL_TERREIN_ID_FIELDNAME);
249             WorkflowStatus originalWorkflowStatus = WorkflowStatus.valueOf(original.getAttribute(WORKFLOW_FIELDNAME).toString());
250 
251             // make a copy of the original feature and set (new) attribute values on the copy
252             SimpleFeature editedNewFeature = createCopy(original);
253             for (int i = 0; i < attributes.size(); i++) {
254                 editedNewFeature.setAttribute(attributes.get(i), values.get(i));
255             }
256 
257             Filter definitief = ff.and(
258                     ff.equals(ff.property(ID_FIELDNAME), ff.literal(original.getAttribute(ID_FIELDNAME))),
259                     ff.equal(ff.property(WORKFLOW_FIELDNAME), ff.literal(WorkflowStatus.definitief.name()), false)
260             );
261             boolean definitiefExists = (this.getStore().getFeatures(definitief).size() > 0);
262 
263             Filter bewerkt = ff.and(
264                     ff.equals(ff.property(ID_FIELDNAME), ff.literal(original.getAttribute(ID_FIELDNAME))),
265                     ff.equal(ff.property(WORKFLOW_FIELDNAME), ff.literal(WorkflowStatus.bewerkt.name()), false)
266             );
267             int aantalBewerkt = this.getStore().getFeatures(bewerkt).size();
268             boolean bewerktExists = (aantalBewerkt > 0);
269 
270             switch (incomingWorkflowStatus) {
271                 case bewerkt:
272                     if (originalWorkflowStatus.equals(WorkflowStatus.definitief)) {
273                         // definitief -> bewerkt
274                         if (bewerktExists) {
275                             // delete existing bewerkt
276                             this.getStore().removeFeatures(bewerkt);
277                         }
278                         // insert new record with original id and workflowstatus "bewerkt", leave original "definitief"
279                         this.getStore().addFeatures(DataUtilities.collection(editedNewFeature));
280                     } else if (originalWorkflowStatus.equals(WorkflowStatus.bewerkt)) {
281                         // bewerkt -> bewerkt, overwrite, only one 'bewerkt' is allowed
282                         if (aantalBewerkt > 1) {
283                             log.error("Er is meer dan 1 bewerkt kavel/terrein voor " + ID_FIELDNAME + "=" + original.getAttribute(ID_FIELDNAME));
284                             // more than 1 bewerkt, move them to archief
285                             this.getStore().modifyFeatures(WORKFLOW_FIELDNAME, WorkflowStatus.archief, bewerkt);
286                         }
287                         this.getStore().modifyFeatures(attributes.toArray(new String[attributes.size()]), values.toArray(), fidFilter);
288                     } else {
289                         // other behaviour not documented eg. archief -> bewerkt, afgevoerd -> bewerkt
290                         //  and not possible to occur in the application as only definitief and bewerkt can be edited
291                         throw new IllegalArgumentException(String.format(
292                                 "Niet ondersteunde workflow stap van %s naar %s",
293                                 originalWorkflowStatus.label(),
294                                 incomingWorkflowStatus.label()));
295                     }
296                     break;
297                 case definitief:
298                     if (definitiefExists) {
299                         // check if any "definitief" exists for this id and move that to "archief"
300                         this.getStore().modifyFeatures(WORKFLOW_FIELDNAME, WorkflowStatus.archief, definitief);
301                     }
302                     if (originalWorkflowStatus.equals(WorkflowStatus.definitief)) {
303                         // if the original was "definitief" insert a new "definitief"
304                         this.getStore().addFeatures(DataUtilities.collection(editedNewFeature));
305                     } else if (originalWorkflowStatus.equals(WorkflowStatus.bewerkt)) {
306                         // if original was "bewerkt" update this to "definitief" with the edits
307                         this.getStore().modifyFeatures(attributes.toArray(new String[attributes.size()]), values.toArray(), fidFilter);
308                     } else {
309                         // other behaviour not documented eg. archief -> definitief, afgevoerd -> definitief
310                         //  and not possible to occur in the application as only definitief and bewerkt can be edited
311                         throw new IllegalArgumentException(String.format(
312                                 "Niet ondersteunde workflow stap van %s naar %s",
313                                 originalWorkflowStatus.label(),
314                                 incomingWorkflowStatus.label()));
315                     }
316                     break;
317                 case afgevoerd:
318                     if (definitiefExists) {
319                         // update any "definitief" for this id to "archief"
320                         this.getStore().modifyFeatures(WORKFLOW_FIELDNAME, WorkflowStatus.archief, definitief);
321                     }
322                     // update original with the new/edited data including "afgevoerd"
323                     this.getStore().modifyFeatures(attributes.toArray(new String[attributes.size()]), values.toArray(), fidFilter);
324 
325                     if (terreinID == null) {
326                         // find any kavels related to this terrein and also set them to "afgevoerd"
327                         Filter kavelFilter = ff.equals(ff.property(KAVEL_TERREIN_ID_FIELDNAME), ff.literal(original.getAttribute(ID_FIELDNAME)));
328                         this.updateKavelWorkflowForTerrein(kavelFilter, WorkflowStatus.afgevoerd);
329                     }
330                     break;
331                 case archief: {
332                     // not described, for now just edit the feature
333                     this.getStore().modifyFeatures(attributes.toArray(new String[attributes.size()]), values.toArray(), fidFilter);
334                     break;
335                 }
336                 default:
337                     throw new IllegalArgumentException("Workflow status van edit feature is null, dit wordt niet ondersteund.");
338             }
339 
340             editTransaction.commit();
341             editTransaction.close();
342             // update terrein geometry
343             if (terreinID != null) {
344                 WorkflowUtil.updateTerreinGeometry(Integer.parseInt(terreinID.toString()), this.getLayer(), incomingWorkflowStatus, this.getApplication(), Stripersist.getEntityManager());
345             }
346         } catch (IllegalArgumentException | IOException | NoSuchElementException e) {
347             editTransaction.rollback();
348             log.error("Ibis editFeature error", e);
349             throw e;
350         }
351     }

353     /**
354      * Make a copy of the original, but with a new fid.
355      *
356      * @param copyFrom original
357      * @return copy having same attribute values as original and a new fid
358      */
359     private SimpleFeature createCopy(SimpleFeature copyFrom) {
360         SimpleFeatureBuilder builder = new SimpleFeatureBuilder(copyFrom.getFeatureType());
361         builder.init(copyFrom);
362         return builder.buildFeature(null);
363     }

365     /**
366      * Update any of the selected kavels to the given workflow.
367      *
368      * @param kavelFilter filter definitie
369      * @param newStatus de nieuwe status
370      */
371     private void updateKavelWorkflowForTerrein(Filter kavelFilter, WorkflowStatus newStatus) {
372         SimpleFeatureStore kavelStore = null;
373         try (Transaction kavelTransaction = new DefaultTransaction("get-related-kavel-geom")) {
374             EntityManager em = Stripersist.getEntityManager();
375             log.debug("updating kavels voor terrein met filter: " + kavelFilter);
376             // find kavel appLayer
377             ApplicationLayer kavelAppLyr = null;
378             List<ApplicationLayer> lyrs = this.getApplication().loadTreeCache(em).getApplicationLayers();
379             for (ListIterator<ApplicationLayer> it = lyrs.listIterator(); it.hasNext();) {
380                 kavelAppLyr = it.next();
381                 if (kavelAppLyr.getLayerName().equalsIgnoreCase(KAVEL_LAYER_NAME)) {
382                     break;
383                 }
384             }
385             Layer l = kavelAppLyr.getService().getLayer(KAVEL_LAYER_NAME, em);
386             kavelStore = (SimpleFeatureStore) l.getFeatureType().openGeoToolsFeatureSource();
387             kavelStore.setTransaction(kavelTransaction);
388             // update kavels
389             kavelStore.modifyFeatures(WORKFLOW_FIELDNAME, newStatus, kavelFilter);
390             kavelTransaction.commit();
391             kavelTransaction.close();
392         } catch (Exception e) {
393             log.error(String.format("Bijwerken van kavel workflow status naar %s voor terrein met %s is mislukt.",
394                     newStatus, kavelFilter), e);
395         } finally {
396             if (kavelStore != null) {
397                 kavelStore.getDataStore().dispose();
398             }
399         }
400     }

402     /**
403      * Check that if {@code disableUserEdit} flag is set on the attribute.
404      * Override superclass behaviour for the workflow field, so that it's not
405      * editable client side, but it can be set programatically in the client.
406      *
407      * @param attrName attribute to check
408      * @return {@code true} when the configured attribute is flagged as
409      * "readOnly" except when this is workflow status
410      */
411     @Override
412     protected boolean isAttributeUserEditingDisabled(String attrName) {
413         boolean isAttributeUserEditingDisabled = super.isAttributeUserEditingDisabled(attrName);
414 
415         if (attrName.equalsIgnoreCase(WORKFLOW_FIELDNAME)) {
416             isAttributeUserEditingDisabled = false;
417         }
418 
419         return isAttributeUserEditingDisabled;
420     }

422     private boolean isSameMutatiedatum(Object datum1, Object datum2) {
423         SimpleDateFormat sdf = new SimpleDateFormat("yyyy-mm-dd");
424         return sdf.format(datum1).equals(sdf.format(datum2));
425     }
