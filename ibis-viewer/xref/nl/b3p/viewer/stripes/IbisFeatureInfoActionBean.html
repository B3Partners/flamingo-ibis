36  /**
37   * Override feature info for {@link KAVEL_LAYER_NAME} and
38   * {@link TERREIN_LAYER_NAME} to make sure we only get to see the current
39   * feature in the client.
40   *
41   * @author mprins
42   */
43  @UrlBinding("/action/ibisfeatureinfo")
44  @StrictBinding
45  public class IbisFeatureInfoActionBean extends FeatureInfoActionBean implements IbisConstants {
46  
47      private static final Log log = LogFactory.getLog(IbisFeatureInfoActionBean.class);
48  
49      /**
50       * execute the query, can be overridden in subclasses to modify behaviour
51       * such as workflow. {@inheritDoc }
52       */
53      @Override
54      protected JSONArray executeQuery(ApplicationLayer al, SimpleFeatureType ft, FeatureSource fs, Query q)
55              throws IOException, JSONException, Exception {
56  
57          JSONArray features;
58          if (this.getLayer().getName().equalsIgnoreCase(KAVEL_LAYER_NAME)
59                  || this.getLayer().getName().equalsIgnoreCase(TERREIN_LAYER_NAME)) {
60  
61              IbisFeatureToJson ftjson = new IbisFeatureToJson(
62                      this.isArrays(),
63                      this.isEdit(),
64                      this.isGraph(),
65                      this.getAttributesToInclude());
66  
67              if (Authorizations.isAppLayerWriteAuthorized(this.getApplication(), al,
68                      this.getContext().getRequest(), Stripersist.getEntityManager())) {
69                  // workflow/edit behaviour
70                  log.debug("Executing custom IBIS featureinfo for write-authorized user on layer " + this.getLayer().getName());
71                  features = ftjson.getWorkflowJSONFeatures(al, this.getLayer().getFeatureType(), fs, q);
72              } else {
73                  log.debug("Executing custom IBIS featureinfo for non-write-authorized user on layer " + this.getLayer().getName());
74                  features = ftjson.getDefinitiefJSONFeatures(al, this.getLayer().getFeatureType(), fs, q);
75              }
76  
77          } else /* not a special layer in this application */ {
78              log.debug("Executing default IBIS featureinfo for 'any' user on layer " + this.getLayer().getName());
79              // default behaviour for any other layers
80              FeatureToJson ftjson = new FeatureToJson(this.isArrays(), this.isEdit(), this.isGraph(), this.getAttributesToInclude());
81              features = ftjson.getJSONFeatures(al, this.getLayer().getFeatureType(), fs, q, null, null);
82          }
83  
84          return features;
85      }
86  
87  }
