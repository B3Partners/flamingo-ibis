64      @Override
65      protected String addNewFeature() throws Exception {
66          String kavelID = super.addNewFeature();
67          //update  terrein
68          Object terreinID = this.getJsonFeature().optString(KAVEL_TERREIN_ID_FIELDNAME, null);
69          WorkflowStatus status = WorkflowStatus.valueOf(this.getJsonFeature().optString(WORKFLOW_FIELDNAME, WorkflowStatus.bewerkt.name()));
70          if (terreinID != null) {
71              WorkflowUtil.updateTerreinGeometry(Integer.parseInt(terreinID.toString()), this.getLayer(), status, this.getApplication(), Stripersist.getEntityManager());
72          }
73          return kavelID;
74      }

76      /**
77       * Override to not delete a feature but set workflow status to
78       * {@code WorkflowStatus.archief}
79       *
80       * @param fid feature id
81       * @throws IOException if any
82       * @throws Exception if any
83       * @see WorkflowStatus
84       */
85      @Override
86      protected void deleteFeature(String fid) throws IOException, Exception {
87          log.debug("ibis deleteFeature: " + fid);
88  
89          Transaction transaction = new DefaultTransaction("ibis_delete");
90          this.getStore().setTransaction(transaction);
91  
92          FilterFactory2 ff = CommonFactoryFinder.getFilterFactory2();
93          Filter filter = ff.id(new FeatureIdImpl(fid));
94  
95          try {
96              this.getStore().modifyFeatures(WORKFLOW_FIELDNAME, WorkflowStatus.afgevoerd, filter);
97              SimpleFeature original = this.getStore().getFeatures(filter).features().next();
98              transaction.commit();
99              Object terreinID = original.getAttribute(KAVEL_TERREIN_ID_FIELDNAME);
100             if (terreinID != null) {
101                 WorkflowUtil.updateTerreinGeometry(Integer.parseInt(terreinID.toString()), this.getLayer(), WorkflowStatus.afgevoerd, this.getApplication(), Stripersist.getEntityManager());
102             }
103         } catch (Exception e) {
104             transaction.rollback();
105             throw e;
106         } finally {
107             transaction.close();
108         }
109     }

111     /**
112      * Override the method from the base class to process our workflow.
113      *
114      * @param fid feature id
115      * @throws Exception if any
116      */
117     @Override
118     protected void editFeature(String fid) throws Exception {
119         log.debug("ibis editFeature:" + fid);
120         FilterFactory2 ff = CommonFactoryFinder.getFilterFactory2();
121         Filter fidFilter = ff.id(new FeatureIdImpl(fid));
122 
123         List<String> attributes = new ArrayList();
124         List values = new ArrayList();
125         WorkflowStatus incomingWorkflowStatus = null;
126         // parse json
127         for (Iterator<String> it = this.getJsonFeature().keys(); it.hasNext();) {
128             String attribute = it.next();
129             if (!this.getFID().equals(attribute)) {
130                 AttributeDescriptor ad = this.getStore().getSchema().getDescriptor(attribute);
131                 if (ad != null) {
132                     if (!isAttributeUserEditingDisabled(attribute)) {
133                         attributes.add(attribute);
134                         if (ad.getType() instanceof GeometryType) {
135                             String wkt = this.getJsonFeature().getString(ad.getLocalName());
136                             Geometry g = null;
137                             if (wkt != null) {
138                                 g = new WKTReader().read(wkt);
139                             }
140                             values.add(g);
141                         } else {
142                             String v = this.getJsonFeature().getString(attribute);
143                             values.add(StringUtils.defaultIfBlank(v, null));
144                             // remember the incoming workflow status
145                             if (attribute.equals(WORKFLOW_FIELDNAME)) {
146                                 incomingWorkflowStatus = WorkflowStatus.valueOf(v);
147                             }
148                         }
149                     } else {
150                         log.info(String.format("Attribute \"%s\" not user editable; ignoring", attribute));
151                     }
152                 } else {
153                     log.warn(String.format("Attribute \"%s\" not in feature type; ignoring", attribute));
154                 }
155             }
156         }
157         log.debug(String.format("Modifying feature source #%d fid=%s, attributes=%s, values=%s",
158                 this.getLayer().getFeatureType().getId(), fid, attributes.toString(), values.toString()));
159 
160         Transaction editTransaction = new DefaultTransaction("ibis_edit");
161         this.getStore().setTransaction(editTransaction);
162         try {
163             if (incomingWorkflowStatus == null) {
164                 throw new IllegalArgumentException("Workflow status van edit feature is null, dit wordt niet ondersteund.");
165             }
166             SimpleFeature original = this.getStore().getFeatures(fidFilter).features().next();
167             Object terreinID = original.getAttribute(KAVEL_TERREIN_ID_FIELDNAME);
168             WorkflowStatus originalWorkflowStatus = WorkflowStatus.valueOf(original.getAttribute(WORKFLOW_FIELDNAME).toString());
169 
170             // make a copy of the original feature and set (new) attribute values on the copy
171             SimpleFeature editedNewFeature = createCopy(original);
172             for (int i = 0; i < attributes.size(); i++) {
173                 editedNewFeature.setAttribute(attributes.get(i), values.get(i));
174             }
175 
176             Filter definitief = ff.and(
177                     ff.equals(ff.property(ID_FIELDNAME), ff.literal(original.getAttribute(ID_FIELDNAME))),
178                     ff.equal(ff.property(WORKFLOW_FIELDNAME), ff.literal(WorkflowStatus.definitief.name()), false)
179             );
180             boolean definitiefExists = (this.getStore().getFeatures(definitief).size() > 0);
181 
182             Filter bewerkt = ff.and(
183                     ff.equals(ff.property(ID_FIELDNAME), ff.literal(original.getAttribute(ID_FIELDNAME))),
184                     ff.equal(ff.property(WORKFLOW_FIELDNAME), ff.literal(WorkflowStatus.bewerkt.name()), false)
185             );
186             int aantalBewerkt = this.getStore().getFeatures(bewerkt).size();
187             boolean bewerktExists = (aantalBewerkt > 0);
188 
189             switch (incomingWorkflowStatus) {
190                 case bewerkt:
191                     if (originalWorkflowStatus.equals(WorkflowStatus.definitief)) {
192                         // definitief -> bewerkt
193                         if (bewerktExists) {
194                             // delete existing bewerkt
195                             this.getStore().removeFeatures(bewerkt);
196                         }
197                         // insert new record with original id and workflowstatus "bewerkt", leave original "definitief"
198                         this.getStore().addFeatures(DataUtilities.collection(editedNewFeature));
199                     } else if (originalWorkflowStatus.equals(WorkflowStatus.bewerkt)) {
200                         // bewerkt -> bewerkt, overwrite, only one 'bewerkt' is allowed
201                         if (aantalBewerkt > 1) {
202                             log.error("Er is meer dan 1 bewerkt kavel/terrein voor " + ID_FIELDNAME + "=" + original.getAttribute(ID_FIELDNAME));
203                             // more than 1 bewerkt, move them to archief
204                             this.getStore().modifyFeatures(WORKFLOW_FIELDNAME, WorkflowStatus.archief, bewerkt);
205                         }
206                         this.getStore().modifyFeatures(attributes.toArray(new String[attributes.size()]), values.toArray(), fidFilter);
207                     } else {
208                         // other behaviour not documented eg. archief -> bewerkt, afgevoerd -> bewerkt
209                         //  and not possible to occur in the application as only definitief and bewerkt can be edited
210                         throw new IllegalArgumentException(String.format(
211                                 "Niet ondersteunde workflow stap van %s naar %s",
212                                 originalWorkflowStatus.label(),
213                                 incomingWorkflowStatus.label()));
214                     }
215                     break;
216                 case definitief:
217                     if (definitiefExists) {
218                         // check if any "definitief" exists for this id and move that to "archief"
219                         this.getStore().modifyFeatures(WORKFLOW_FIELDNAME, WorkflowStatus.archief, definitief);
220                     }
221                     if (originalWorkflowStatus.equals(WorkflowStatus.definitief)) {
222                         // if the original was "definitief" insert a new "definitief"
223                         this.getStore().addFeatures(DataUtilities.collection(editedNewFeature));
224                     } else if (originalWorkflowStatus.equals(WorkflowStatus.bewerkt)) {
225                         // if original was "bewerkt" update this to "definitief" with the edits
226                         this.getStore().modifyFeatures(attributes.toArray(new String[attributes.size()]), values.toArray(), fidFilter);
227                     } else {
228                         // other behaviour not documented eg. archief -> definitief, afgevoerd -> definitief
229                         //  and not possible to occur in the application as only definitief and bewerkt can be edited
230                         throw new IllegalArgumentException(String.format(
231                                 "Niet ondersteunde workflow stap van %s naar %s",
232                                 originalWorkflowStatus.label(),
233                                 incomingWorkflowStatus.label()));
234                     }
235                     break;
236                 case afgevoerd:
237                     if (definitiefExists) {
238                         // update any "definitief" for this id to "archief"
239                         this.getStore().modifyFeatures(WORKFLOW_FIELDNAME, WorkflowStatus.archief, definitief);
240                     }
241                     // update original with the new/edited data including "afgevoerd"
242                     this.getStore().modifyFeatures(attributes.toArray(new String[attributes.size()]), values.toArray(), fidFilter);
243 
244                     if (terreinID == null) {
245                         // find any kavels related to this terrein and also set them to "afgevoerd"
246                         Filter kavelFilter = ff.equals(ff.property(KAVEL_TERREIN_ID_FIELDNAME), ff.literal(original.getAttribute(ID_FIELDNAME)));
247                         this.updateKavelWorkflowForTerrein(kavelFilter, WorkflowStatus.afgevoerd);
248                     }
249                     break;
250                 case archief: {
251                     // not described, for now just edit the feature
252                     this.getStore().modifyFeatures(attributes.toArray(new String[attributes.size()]), values.toArray(), fidFilter);
253                     break;
254                 }
255                 default:
256                     throw new IllegalArgumentException("Workflow status van edit feature is null, dit wordt niet ondersteund.");
257             }
258 
259             editTransaction.commit();
260             editTransaction.close();
261             // update terrein geometry
262             if (terreinID != null) {
263                 WorkflowUtil.updateTerreinGeometry(Integer.parseInt(terreinID.toString()), this.getLayer(), incomingWorkflowStatus, this.getApplication(), Stripersist.getEntityManager());
264             }
265         } catch (IllegalArgumentException | IOException | NoSuchElementException e) {
266             editTransaction.rollback();
267             log.error("Ibis editFeature error", e);
268             throw e;
269         }
270     }

272     /**
273      * Make a copy of the original, but with a new fid.
274      *
275      * @param copyFrom original
276      * @return copy having same attribute values as original and a new fid
277      */
278     private SimpleFeature createCopy(SimpleFeature copyFrom) {
279         SimpleFeatureBuilder builder = new SimpleFeatureBuilder(copyFrom.getFeatureType());
280         builder.init(copyFrom);
281         return builder.buildFeature(null);
282     }

284     /**
285      * Update any of the selected kavels to the given workflow.
286      *
287      * @param kavelFilter filter definitie
288      * @param newStatus de nieuwe status
289      */
290     private void updateKavelWorkflowForTerrein(Filter kavelFilter, WorkflowStatus newStatus) {
291         SimpleFeatureStore kavelStore = null;
292         try (Transaction kavelTransaction = new DefaultTransaction("get-related-kavel-geom")) {
293             EntityManager em = Stripersist.getEntityManager();
294             log.debug("updating kavels voor terrein met filter: " + kavelFilter);
295             // find kavel appLayer
296             ApplicationLayer kavelAppLyr = null;
297             List<ApplicationLayer> lyrs = this.getApplication().loadTreeCache(em).getApplicationLayers();
298             for (ListIterator<ApplicationLayer> it = lyrs.listIterator(); it.hasNext();) {
299                 kavelAppLyr = it.next();
300                 if (kavelAppLyr.getLayerName().equalsIgnoreCase(KAVEL_LAYER_NAME)) {
301                     break;
302                 }
303             }
304             Layer l = kavelAppLyr.getService().getLayer(KAVEL_LAYER_NAME, em);
305             kavelStore = (SimpleFeatureStore) l.getFeatureType().openGeoToolsFeatureSource();
306             kavelStore.setTransaction(kavelTransaction);
307             // update kavels
308             kavelStore.modifyFeatures(WORKFLOW_FIELDNAME, newStatus, kavelFilter);
309             kavelTransaction.commit();
310             kavelTransaction.close();
311         } catch (Exception e) {
312             log.error(String.format("Bijwerken van kavel workflow status naar %s voor terrein met %s is mislukt.",
313                     newStatus, kavelFilter), e);
314         } finally {
315             if (kavelStore != null) {
316                 kavelStore.getDataStore().dispose();
317             }
318         }
319     }

321     /**
322      * Check that if {@code disableUserEdit} flag is set on the attribute.
323      * Override superclass behaviour for the workflow field, so that it's not
324      * editable client side, but it can be set programatically in the client.
325      *
326      * @param attrName attribute to check
327      * @return {@code true} when the configured attribute is flagged as
328      * "readOnly" except when this is workflow status
329      */
330     @Override
331     protected boolean isAttributeUserEditingDisabled(String attrName) {
332         boolean isAttributeUserEditingDisabled = super.isAttributeUserEditingDisabled(attrName);
333 
334         if (attrName.equalsIgnoreCase(WORKFLOW_FIELDNAME)) {
335             isAttributeUserEditingDisabled = false;
336         }
337 
338         return isAttributeUserEditingDisabled;
339     }

341     private boolean isSameMutatiedatum(Object datum1, Object datum2) {
342         SimpleDateFormat sdf = new SimpleDateFormat("yyyy-mm-dd");
343         return sdf.format(datum1).equals(sdf.format(datum2));
344     }
