27  import java.util.ListIterator;

29  import nl.b3p.viewer.config.app.Application;
30  import nl.b3p.viewer.config.app.ApplicationLayer;

32  import org.apache.commons.logging.Log;
33  import org.apache.commons.logging.LogFactory;

35  import org.geotools.data.Transaction;
36  import org.geotools.data.simple.SimpleFeatureCollection;
37  import org.geotools.data.simple.SimpleFeatureStore;
38  import org.geotools.factory.CommonFactoryFinder;
39  import org.geotools.feature.collection.AbstractFeatureVisitor;
40  import org.geotools.geometry.jts.JTSFactoryFinder;
41  import org.opengis.feature.Feature;
42  import org.opengis.filter.Filter;
43  import org.opengis.filter.FilterFactory2;

45  /**
46   * Utility method that come in handy hadling workflow.
47   *
48   * @author mprins
49   */
50  public class WorkflowUtil implements IbisConstants {
51  
52      private static final Log log = LogFactory.getLog(WorkflowUtil.class);
53  
54      /**
55       * private constructor for utility class.
56       */
57      private WorkflowUtil() {
58      }
59  
60      /**
61       * Update the geometry of the TERREIN. Must be called after the kavels
62       * transaction. Voor definitief terrein definitief kavels gebruiken, voor
63       * bewerkt terrein definitief en bewerkt kavel gebruiken.
64       * <p>
65       * Als een terrein "per ongeluk" is afgevoerd moet het terugkomen mits kavel definitief is
66       * De meest recente terrein versie - per definitie (workflow) is dat het "afgevoerde" - moet dan definitief worden.
67       *
68       * @param terreinID   feature id van het terrein
69       * @param layer       kavels layer
70       * @param kavelStatus workflow status van te gebruiken kavels
71       * @param application flamingo applicatie
72       * @param em          persistence manager
73       */
74      public static void updateTerreinGeometry(Integer terreinID, Layer layer, WorkflowStatus kavelStatus, Application application, EntityManager em) {
75          log.debug("Updating terrein geometry for " + terreinID);
76          SimpleFeatureStore terreinStore = null;
77          SimpleFeatureStore kavelStore = null;
78          FilterFactory2 ff = CommonFactoryFinder.getFilterFactory2();
79          Transaction terreinTransaction = new DefaultTransaction("edit-terrein-geom");
80          Transaction kavelTransaction = new DefaultTransaction("get-kavel-geom");
81          try {
82              // determine whichs kavels to use for calcutating new geometry
83              Filter kavelFilter = Filter.EXCLUDE;
84              switch (kavelStatus) {
85                  case bewerkt:
86                      // find all "definitief" and "bewerkt" kavel for terreinID
87                      kavelFilter = ff.and(
88                              ff.equals(ff.property(KAVEL_TERREIN_ID_FIELDNAME), ff.literal(terreinID)),
89                              ff.or(
90                                      ff.equal(ff.property(WORKFLOW_FIELDNAME), ff.literal(WorkflowStatus.bewerkt.toString()), false),
91                                      ff.equal(ff.property(WORKFLOW_FIELDNAME), ff.literal(WorkflowStatus.definitief.toString()), false)
92                              ));
93                      break;
94                  case archief:
95                  case afgevoerd:
96                  case definitief:
97                      // find all "definitief" kavel for terreinID
98                      kavelFilter = ff.and(
99                              ff.equals(ff.property(KAVEL_TERREIN_ID_FIELDNAME), ff.literal(terreinID)),
100                             ff.equal(ff.property(WORKFLOW_FIELDNAME), ff.literal(WorkflowStatus.definitief.toString()), false)
101                     );
102                     break;
103                 default:
104                 // do nothing / should not happen
105             }
106             log.debug("Looking for kavel(s) with filter: " + kavelFilter);
107 
108             kavelStore = (SimpleFeatureStore) layer.getFeatureType().openGeoToolsFeatureSource();
109             kavelStore.setTransaction(kavelTransaction);
110             SimpleFeatureCollection kavels = kavelStore.getFeatures(kavelFilter);
111 
112             // dissolve all kavel geometries
113             final ArrayList<Geometry> kavelGeoms = new ArrayList();
114             kavels.accepts(new AbstractFeatureVisitor() {
115                 @Override
116                 public void visit(Feature feature) {
117                     Geometry geom = (Geometry) feature.getDefaultGeometryProperty().getValue();
118                     if (geom != null) {
119                         kavelGeoms.add(geom);
120                     }
121                 }
122             }, null);
123             log.debug("Kavels found for this terrein: " + kavelGeoms.size());
124 
125             Geometry newTerreinGeom;
126             if (kavelGeoms.size() == 1) {
127                 // maar 1 (vlak) geom gevonden, dus geom overzetten naar terrein
128                 newTerreinGeom = kavelGeoms.get(0);
129             } else {
130                 GeometryFactory factory = JTSFactoryFinder.getGeometryFactory(null);
131                 GeometryCollection geometryCollection = (GeometryCollection) factory.buildGeometry(kavelGeoms);
132                 newTerreinGeom = geometryCollection.union();
133             }
134             // buffer + en -0.001m (because rijksdriehoek) om interne slivers in terrein op te lossen,
135             // zie https://github.com/B3Partners/flamingo-ibis/issues/64
136             log.debug("terrein geom : " + newTerreinGeom);
137             newTerreinGeom = newTerreinGeom.buffer(.001, 1);
138             log.trace("buffer+.001   : " + newTerreinGeom);
139             newTerreinGeom = newTerreinGeom.buffer(-.001, 1);
140             log.trace("buffer-.001   : " + newTerreinGeom);
141             // en ook nog een snap-to-self met een centimeter tolerantie
142             newTerreinGeom = GeometrySnapper.snapToSelf(newTerreinGeom, .01, true);
143             log.debug("snapped+clean: " + newTerreinGeom);
144 
145             if (!newTerreinGeom.getGeometryType().equalsIgnoreCase("MultiPolygon")) {
146                 GeometryFactory f = JTSFactoryFinder.getGeometryFactory();
147                 newTerreinGeom = f.createMultiPolygon(new Polygon[]{(Polygon) newTerreinGeom});
148             }
149 
150             // find terrein appLayer
151             ApplicationLayer terreinAppLyr = null;
152             List<ApplicationLayer> lyrs = application.loadTreeCache(em).getApplicationLayers();
153             for (ListIterator<ApplicationLayer> it = lyrs.listIterator(); it.hasNext();) {
154                 terreinAppLyr = it.next();
155                 if (terreinAppLyr.getLayerName().equalsIgnoreCase(TERREIN_LAYER_NAME)) {
156                     break;
157                 }
158             }
159             Layer l = terreinAppLyr.getService().getLayer(TERREIN_LAYER_NAME, em);
160             terreinStore = (SimpleFeatureStore) l.getFeatureType().openGeoToolsFeatureSource();
161             terreinStore.setTransaction(terreinTransaction);
162 
163             // determine which terrein to update
164             Filter terreinFilter = Filter.EXCLUDE;
165             switch (kavelStatus) {
166                 case bewerkt:
167                     terreinFilter = ff.and(
168                             ff.equals(ff.property(ID_FIELDNAME), ff.literal(terreinID)),
169                             ff.equal(ff.property(WORKFLOW_FIELDNAME), ff.literal(WorkflowStatus.bewerkt.name()), false)
170                     );
171                     break;
172                 case archief:
173                 case afgevoerd:
174                 case definitief:
175                     terreinFilter = ff.and(
176                             ff.equals(ff.property(ID_FIELDNAME), ff.literal(terreinID)),
177                             ff.equal(ff.property(WORKFLOW_FIELDNAME), ff.literal(WorkflowStatus.definitief.name()), false)
178                     );
179                     break;
180                 default:
181                 // won't do anything
182             }
183 
184             if (kavelStatus.equals(WorkflowStatus.definitief)) {
185                 // check of terrein "per ongeluk" afgevoerd, maar kavel definitief
186                 final Filter perOngelukFilter = ff.and(
187                         ff.equals(ff.property(ID_FIELDNAME), ff.literal(terreinID)),
188                         ff.equal(ff.property(WORKFLOW_FIELDNAME), ff.literal(WorkflowStatus.afgevoerd.name()), false)
189                 );
190                 log.debug("Herstel 'per ongeluk' 'afgevoerd' terrein naar 'definitief' met filter: " + terreinFilter);
191                 terreinStore.modifyFeatures(WORKFLOW_FIELDNAME, WorkflowStatus.definitief.name(), perOngelukFilter);
192             }
193 
194             // update terrein with new geom
195             log.debug("Update terrein geom for kavels filtered by: " + terreinFilter);
196             String geomAttrName = terreinStore.getSchema().getGeometryDescriptor().getLocalName();
197             terreinStore.modifyFeatures(geomAttrName, newTerreinGeom, terreinFilter);
198             terreinTransaction.commit();
199             log.debug("Done updating terrein geometry for " + terreinFilter + " with new geom: " + newTerreinGeom + " on attribute " + geomAttrName);
200         } catch (Exception e) {
201             log.error(String.format("Update van terrein geometrie %s is mislukt", terreinID), e);
202         } finally {
203             try {
204                 terreinTransaction.close();
205             } catch (IOException io) {
206                 log.warn("Error closing terrein transaction", io);
207             }
208             if (terreinStore != null) {
209                 terreinStore.getDataStore().dispose();
210             }
211             try {
212                 kavelTransaction.close();
213             } catch (IOException io) {
214                 log.warn("Error closing kavel transaction", io);
215             }
216             if (kavelStore != null) {
217                 kavelStore.getDataStore().dispose();
218             }
219         }
220     }
221 }
